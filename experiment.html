                </li>

                <li><strong>Trials:</strong> 10 trials per condition (20 total)</li>

                <li><strong>Timing:</strong> "BRAKE" will appear randomly between 2-6 seconds</li>

    <script>

        // Global variables

        let currentCondition = 0; // 0 = no texting, 1 = texting

        let conditionOrder = []; // Will be randomized

        let conditionIndex = 0; // Index in the randomized order

        let currentTrial = 0;

        let trialsPerCondition = 10;

        let reactionTimes = { noTexting: [], texting: [] };

        let stimulusStartTime = 0;

        let stimulusTimeout = null;

        let waitingForResponse = false;

        let conditionStarted = false;

        let betweenConditions = false;

        

        const textMessage = "The quick brown fox jumps over the lazy dog";

        

        // Main experiment functions

        function startExperiment() {

            document.querySelector('.instructions').style.display = 'none';

            document.querySelector('.start-button').style.display = 'none';

            document.getElementById('experimentArea').style.display = 'block';

            

            // Randomize condition order

            conditionOrder = Math.random() < 0.5 ? [0, 1] : [1, 0];

            conditionIndex = 0;

            currentTrial = 0;

            reactionTimes = { noTexting: [], texting: [] };

            betweenConditions = false;

            

            currentCondition = conditionOrder[conditionIndex];

            startCondition();

        }

        

        function startCondition() {

            const conditionNames = ['No Texting Condition', 'Texting Condition'];

            const conditionClasses = ['no-texting', 'texting'];

            

            document.getElementById('conditionInfo').textContent = conditionNames[currentCondition];

            document.getElementById('conditionInfo').className = `condition-info ${conditionClasses[currentCondition]}`;

            

            // Show texting instructions for texting condition

            if (currentCondition === 1) {

                document.getElementById('textMessage').style.display = 'block';

            } else {

                document.getElementById('textMessage').style.display = 'none';

            }

            

            currentTrial = 0;

            conditionStarted = false;

            

            // Countdown before starting condition

            let countdown = 5;

            document.getElementById('countdown').textContent = `Starting in ${countdown} seconds...`;

            

            const countdownInterval = setInterval(() => {

                countdown--;

                if (countdown > 0) {

                    document.getElementById('countdown').textContent = `Starting in ${countdown} seconds...`;

                } else {

                    document.getElementById('countdown').textContent = '';

                    clearInterval(countdownInterval);

                    conditionStarted = true;

                    startTrial();

                }

            }, 1000);

        }

        

        function startTrial() {

            if (!conditionStarted) return;

            

            currentTrial++;

            document.getElementById('trialInfo').textContent = 

                `Trial ${currentTrial} of ${trialsPerCondition} - ${currentCondition === 0 ? 'No Texting' : 'Texting'}`;

            

            document.getElementById('waitingMessage').textContent = 'Get Ready...';

            document.getElementById('waitingMessage').style.display = 'block';

            document.getElementById('earlyResponse').style.display = 'none';

            

            // Clear any existing brake word

            const existingBrake = document.querySelector('.brake-word');

            if (existingBrake) {

                existingBrake.remove();

            }

            

            waitingForResponse = false;

            

            // Random delay between 2-6 seconds

            const delay = Math.random() * 4000 + 2000;

            

            stimulusTimeout = setTimeout(() => {

                showBrakeStimulus();

            }, delay);

        }

        

        function showBrakeStimulus() {

            document.getElementById('waitingMessage').style.display = 'none';

            

            // Create brake word element

            const brakeWord = document.createElement('div');

            brakeWord.className = 'brake-word';

            brakeWord.textContent = 'BRAKE';

            

            // Random position within the stimulus area

            // Account for the brake word size (approximately 200px wide, 80px tall)

            const maxX = 566 - 200;

            const maxY = 566 - 80;

            const randomX = Math.random() * maxX;

            const randomY = Math.random() * maxY;

            

            brakeWord.style.left = randomX + 'px';

            brakeWord.style.top = randomY + 'px';

            

            document.getElementById('stimulusArea').appendChild(brakeWord);

            

            stimulusStartTime = performance.now();

            waitingForResponse = true;

        }

        

        function handleResponse() {

            if (!conditionStarted || betweenConditions) return;

            

            if (!waitingForResponse) {

                // Early response

                document.getElementById('earlyResponse').style.display = 'block';

                clearTimeout(stimulusTimeout);

                setTimeout(() => {

                    startTrial();

                }, 1500);

                return;

            }

            

            // Calculate reaction time

            const reactionTime = performance.now() - stimulusStartTime;

            

            // Store reaction time

            if (currentCondition === 0) {

                reactionTimes.noTexting.push(reactionTime);

            } else {

                reactionTimes.texting.push(reactionTime);

            }

            

            // Clear stimulus

            const brakeWord = document.querySelector('.brake-word');

            if (brakeWord) {

                brakeWord.remove();

            }

            

            waitingForResponse = false;

            

            // Check if condition is complete

            if (currentTrial >= trialsPerCondition) {

                if (conditionIndex === 0) {

                    // Move to next condition with gap

                    showBetweenConditionsMessage();

                } else {

                    // Experiment complete

                    showResults();

                }

            } else {

                // Next trial

                setTimeout(() => {

                    startTrial();

                }, 1000);

            }

        }

        

        function showBetweenConditionsMessage() {

            document.getElementById('experimentArea').style.display = 'none';

            document.getElementById('betweenConditions').style.display = 'block';

            

            // Set up instructions for next condition

            conditionIndex = 1;

            const nextCondition = conditionOrder[conditionIndex];

            const nextConditionInstructions = document.getElementById('nextConditionInstructions');

            

            if (nextCondition === 0) {

                // Next is no texting

                nextConditionInstructions.innerHTML = `

                    <p><strong>Next Condition: No Texting</strong></p>

                    <p>In this condition, you will:</p>

                    <ul style="text-align: left; display: inline-block;">

                        <li>Focus only on watching for the word "BRAKE"</li>

                        <li>Press SPACEBAR as quickly as possible when you see it</li>

                        <li>Put your phone away - no texting required</li>

                    </ul>

                `;

            } else {

                // Next is texting

                nextConditionInstructions.innerHTML = `

                    <p><strong>Next Condition: Texting</strong></p>

                    <p>In this condition, you will:</p>

                    <ul style="text-align: left; display: inline-block;">

                        <li><strong>Get your phone ready now!</strong></li>

                        <li>Type the message: "${textMessage}"</li>

                        <li>While texting, also watch for the word "BRAKE" and press SPACEBAR when you see it</li>

                        <li>Try to do both tasks at the same time</li>

                    </ul>

                `;

            }

            

            betweenConditions = true;

        }

        

        function continueToNextCondition() {

            document.getElementById('betweenConditions').style.display = 'none';

            document.getElementById('experimentArea').style.display = 'block';

            

            currentCondition = conditionOrder[conditionIndex];

            betweenConditions = false;

            startCondition();

        }

        

        function showResults() {

            document.getElementById('experimentArea').style.display = 'none';

            document.getElementById('results').style.display = 'block';

            

            // Calculate means

            const noTextingMean = reactionTimes.noTexting.reduce((a, b) => a + b, 0) / reactionTimes.noTexting.length;

            const textingMean = reactionTimes.texting.reduce((a, b) => a + b, 0) / reactionTimes.texting.length;

            const difference = textingMean - noTextingMean;

            const percentIncrease = ((difference / noTextingMean) * 100);

            

            let html = `

                <div class="result-item no-texting-result">

                    <strong>No Texting Condition:</strong><br>

                    Mean Reaction Time: ${noTextingMean.toFixed(0)} milliseconds

                </div>

                <div class="result-item texting-result">

                    <strong>Texting Condition:</strong><br>

                    Mean Reaction Time: ${textingMean.toFixed(0)} milliseconds

                </div>

                <div class="difference">

                    <strong>Difference:</strong> ${difference.toFixed(0)} milliseconds<br>

                    <strong>Percent Increase:</strong> ${percentIncrease.toFixed(1)}%<br>

                    ${difference > 0 ? 'Texting INCREASED reaction time' : 'Texting DECREASED reaction time'}

                </div>

                <div style="margin-top: 20px; text-align: left;">

                    <h3>Individual Trial Data:</h3>

                    <strong>No Texting:</strong> ${reactionTimes.noTexting.map(rt => rt.toFixed(0) + 'ms').join(', ')}<br><br>

                    <strong>Texting:</strong> ${reactionTimes.texting.map(rt => rt.toFixed(0) + 'ms').join(', ')}

                </div>

            `;

            

            document.getElementById('resultDetails').innerHTML = html;

        }

        

        function restart() {

            document.getElementById('results').style.display = 'none';

            document.getElementById('betweenConditions').style.display = 'none';

            document.querySelector('.instructions').style.display = 'block';

            document.querySelector('.start-button').style.display = 'inline-block';

            

            // Reset all variables

            currentCondition = 0;

            conditionOrder = [];

            conditionIndex = 0;

            currentTrial = 0;

            reactionTimes = { noTexting: [], texting: [] };

            waitingForResponse = false;

            conditionStarted = false;

            betweenConditions = false;

            

            if (stimulusTimeout) {

                clearTimeout(stimulusTimeout);

            }

        }

        

        // Event listeners

        document.addEventListener('keydown', function(event) {

            if (event.code === 'Space' && !betweenConditions) {

                event.preventDefault();

                handleResponse();

            }

        });

        

        // Prevent page scrolling with spacebar

        document.addEventListener('keydown', function(event) {

            if (event.code === 'Space') {

                event.preventDefault();

            }

        });

    </script>
